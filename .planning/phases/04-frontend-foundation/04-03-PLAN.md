**Step 1: Update `frontend/next.config.ts` to add streaming-friendly headers**

Next.js can buffer proxied responses. To ensure SSE streams through rewrites without buffering, add a `headers()` configuration alongside the existing `rewrites()`. This tells any upstream proxy (Nginx, etc.) to not buffer the response, and sets `Cache-Control: no-store` for the SSE endpoint:

```typescript
import type { NextConfig } from "next";

const nextConfig: NextConfig = {
  async headers() {
    return [
      {
        source: "/api/stream/:path*",
        headers: [
          {
            key: "X-Accel-Buffering",
            value: "no",
          },
          {
            key: "Cache-Control",
            value: "no-store",
          },
          {
            key: "Content-Type",
            value: "text/event-stream",
          },
        ],
      },
    ];
  },
  async rewrites() {
    return [
      {
        source: "/api/:path*",
        destination: `${process.env.BACKEND_URL || "http://localhost:8000"}/api/:path*`,
      },
    ];
  },
};

export default nextConfig;
```

```typescript
"use client";

import { useEffect, useRef } from "react";
import { usePriceStore } from "@/stores/priceStore";

export function useSSE() {
  const updatePrices = usePriceStore((s) => s.updatePrices);
  const setConnectionStatus = usePriceStore((s) => s.setConnectionStatus);
  const eventSourceRef = useRef<EventSource | null>(null);

  useEffect(() => {
    const es = new EventSource("/api/stream/prices");
    eventSourceRef.current = es;

    es.onopen = () => {
      setConnectionStatus("connected");
    };

    es.onmessage = (event) => {
      try {
        const data = JSON.parse(event.data);
        updatePrices(data);
      } catch {
        // Ignore malformed SSE data
      }
    };

    es.onerror = () => {
      if (es.readyState === EventSource.CONNECTING) {
        setConnectionStatus("reconnecting");
      } else {
        setConnectionStatus("disconnected");
      }
    };

    return () => {
      es.close();
      setConnectionStatus("disconnected");
    };
  }, [updatePrices, setConnectionStatus]);
}
```

**Step 1: Add price history accumulation to `frontend/src/stores/priceStore.ts`**

Add a `priceHistory` field to the price store that accumulates price points per ticker from SSE data. This is the data source for sparklines.

Modify the store to add:
```typescript
import { create } from "zustand";
import type { PriceUpdate } from "@/types/api";

export type ConnectionStatus = "connected" | "reconnecting" | "disconnected";

const MAX_HISTORY_POINTS = 50; // Keep last 50 data points per ticker for sparklines

interface PriceStore {
  prices: Record<string, PriceUpdate>;
  priceHistory: Record<string, number[]>;  // NEW: ticker -> array of prices
  connectionStatus: ConnectionStatus;
  updatePrices: (data: Record<string, PriceUpdate>) => void;
  appendPriceHistory: (data: Record<string, PriceUpdate>) => void;  // NEW
  setConnectionStatus: (status: ConnectionStatus) => void;
}

export const usePriceStore = create<PriceStore>()((set) => ({
  prices: {},
  priceHistory: {},  // NEW
  connectionStatus: "disconnected",
  updatePrices: (data) =>
    set((state) => ({ prices: { ...state.prices, ...data } })),
  appendPriceHistory: (data) =>  // NEW
    set((state) => {
      const newHistory = { ...state.priceHistory };
      for (const [ticker, update] of Object.entries(data)) {
        const existing = newHistory[ticker] || [];
        const updated = [...existing, update.price];
        // Cap at MAX_HISTORY_POINTS to prevent unbounded growth
        newHistory[ticker] = updated.length > MAX_HISTORY_POINTS
          ? updated.slice(updated.length - MAX_HISTORY_POINTS)
          : updated;
      }
      return { priceHistory: newHistory };
    }),
  setConnectionStatus: (connectionStatus) => set({ connectionStatus }),
}));
```

Key details:
- `priceHistory` is `Record<string, number[]>` -- just raw prices, not full PriceUpdate objects (sparklines only need the price value).
- `appendPriceHistory` adds the latest price for each ticker to the history array.
- Cap at 50 data points per ticker (`MAX_HISTORY_POINTS`). This gives enough data for a meaningful sparkline without unbounded memory growth. At ~500ms updates, this represents ~25 seconds of data. As more data accumulates, old points are dropped from the front.
- Export the constant `MAX_HISTORY_POINTS` is not needed externally.

**Step 3: Create sparkline component** `frontend/src/components/watchlist/Sparkline.tsx`

Build a simple inline SVG sparkline. No external library needed -- this is ~40 lines of code for a polyline SVG.

```typescript
"use client";

interface SparklineProps {
  data: number[];
  width?: number;
  height?: number;
  color?: string;
}

export function Sparkline({
  data,
  width = 60,
  height = 20,
  color = "#209dd7",
}: SparklineProps) {
  if (data.length < 2) {
    // Not enough data points yet -- show a thin muted line placeholder
    return (
      <svg width={width} height={height} className="flex-shrink-0">
        <line
          x1={0}
          y1={height / 2}
          x2={width}
          y2={height / 2}
          stroke="#6e7681"
          strokeWidth={1}
          strokeDasharray="2,2"
        />
      </svg>
    );
  }

  const min = Math.min(...data);
  const max = Math.max(...data);
  const range = max - min || 1; // Avoid division by zero for flat prices

  // Determine color based on trend: green if last > first, red if last < first, blue if flat
  const trendColor =
    data[data.length - 1] > data[0]
      ? "#3fb950"  // price-up green
      : data[data.length - 1] < data[0]
        ? "#f85149"  // price-down red
        : color;     // default blue for flat

  const padding = 2;
  const chartWidth = width - padding * 2;
  const chartHeight = height - padding * 2;

  const points = data
    .map((value, index) => {
      const x = padding + (index / (data.length - 1)) * chartWidth;
      const y = padding + chartHeight - ((value - min) / range) * chartHeight;
      return `${x.toFixed(1)},${y.toFixed(1)}`;
    })
    .join(" ");

  return (
    <svg width={width} height={height} className="flex-shrink-0">
      <polyline
        points={points}
        fill="none"
        stroke={trendColor}
        strokeWidth={1.5}
        strokeLinecap="round"
        strokeLinejoin="round"
      />
    </svg>
  );
}
```

Key design decisions:
- **Pure SVG, no canvas or external library.** A polyline is all a sparkline needs. This is the lightest possible approach.
- **Color reflects trend:** Green if price went up since first data point, red if down, blue default if flat. Matches the trading terminal aesthetic.
- **Graceful empty state:** When fewer than 2 data points exist (early page load), shows a thin dashed line placeholder. This prevents layout shifts and communicates "filling in".
- **Capped dimensions:** Default 60x20px fits in the watchlist row without disrupting the dense layout.
- **Padding:** 2px padding avoids SVG clipping on the polyline edges.
